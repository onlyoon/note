#### 10430번

fs 모듈 사용
```js
const fs = require("fs");
const input = fs.readFileSync("dev/stdin").toString().trim().split(' ').map(v=>Number(v));
const [a, b, c] = input;
console.log((a+b)%c);
console.log(((a%c)+(b%c))% c);
console.log((a*b)%c);
console.log(((a%c)*(b%c))%c);
```
