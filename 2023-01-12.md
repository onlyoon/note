#### 10926번

readline 모듈 활용

```js
const readline = require("readline");

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

rl.on("line", (line) => {
    const input = line;
    console.log(input+'??!');
    rl.close();
})

rl.on("close", () => {
    process.exit();
})
```
